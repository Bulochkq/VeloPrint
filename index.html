/**
 * Velocity Service Printing Bookmarklet
 * 
 * This bookmarklet extracts service order details from the active CRM/service page
 * and generates a beautifully formatted, print-optimized confirmation sheet.
 * It supports double-printing (two copies on a single A4 page for small orders)
 * and single-printing (one large copy for orders with more than 9 items).
 */
(function () {
  // 1. Extract Order & Booking Details
  let orderNo = document.querySelector('input[name="cislo"]')?.value || '';
  let resNo = document.querySelector('input[name="cislo_rezervacie"]')?.value || '';
  let bike = document.querySelector('input[name="nazov_bike"]')?.value || '';
  let customer = document.getElementById('odberatel')?.value || '';
  
  // Parse Customer Info (Phone and Email are combined in a text block)
  let infoText = document.getElementById('odberatel_info')?.innerText || '';
  let phone = (infoText.match(/Telefon:\s*([\+\d\s]+)/) || [])[1]?.trim() || '';
  let email = (infoText.match(/E-mail:\s*([^\s]+)/) || [])[1]?.trim() || '';
  
  // Check service status
  let stavSel = document.getElementById('id_stav');
  let isSpravene = stavSel && stavSel.options[stavSel.selectedIndex].text === 'Spravené';
  
  // Parse Dates
  let datumPrijatia = `${document.getElementById('vyhotovenie_den')?.value || ''}.${document.getElementById('vyhotovenie_mesiac')?.value || ''}.${document.getElementById('vyhotovenie_rok')?.value || ''}`;
  let datumOdovzdania = `${document.getElementById('splatnost_den')?.value || ''}.${document.getElementById('splatnost_mesiac')?.value || ''}.${document.getElementById('splatnost_rok')?.value || ''}`;
  
  // Parse Total Price
  let sumaZakaznik = document.getElementById('suma_zakaznik')?.value || '';
  if (sumaZakaznik === '0' || sumaZakaznik === '0.00' || sumaZakaznik === '0,00') {
    sumaZakaznik = '';
  }

  // 2. Extract Service items (Job description & parts pricing)
  let items = [];
  let estL = 0; // Estimated line height/complexity count
  
  document.querySelectorAll('[id^="input_polozka_riadok_"]').forEach(r => {
    // Only process visible lines
    if (r.style.display !== 'none') {
      let m = r.id.match(/_(\d+)$/);
      if (m) {
        let i = m[1];
        let n = document.querySelector(`input[name="polozka[${i}][nazov]"]`)?.value || '';
        let ps = document.querySelector(`input[name="polozka[${i}][cena_servis]"]`)?.value || '';
        let pd = document.querySelector(`input[name="polozka[${i}][cena_jed]"]`)?.value || '';
        
        if (n.trim() || ps.trim() || pd.trim()) {
          items.push({ name: n, priceSrv: ps, priceDiely: pd });
          // Estimate print lines needed based on text length (55 chars per line limit)
          estL += Math.max(1, Math.ceil(n.length / 55));
        }
      }
    }
  });

  // Decide if we print a single full-page ticket or a double ticket (2 copies on one A4)
  let sgl = estL > 9 || items.length > 9;
  let tgL = sgl ? Math.max(items.length, 25) : 9; // Target lines (fill empty rows)
  
  // Fill remaining items with empty structures to maintain consistent height
  while (items.length < tgL) {
    items.push({ name: '', priceSrv: '', priceDiely: '' });
  }

  // 3. HTML Template Generator
  const gen = (s) => `
<div style="display: flex; flex-direction: row; width: 100%; height: ${s ? '98vh' : '48vh'}; box-sizing: border-box; margin-bottom: 0; ${s ? 'padding-bottom: 1px;' : ''}">
  <div style="font-family: Arial, sans-serif; display: flex; flex-direction: column; flex-grow: 1; height: 100%; box-sizing: border-box; color: #000; border: 2px solid #000; border-radius: 12px; overflow: hidden;">
    <table style="width: 100%; height: 100%; flex-grow: 1; border-collapse: separate; border-spacing: 0; table-layout: fixed; background: #fff;">
      <colgroup>
        <col style="width: 42%;">
        <col style="width: 10%;">
        <col style="width: 10%;">
        <col style="width: 38%;">
      </colgroup>
      
      <!-- Table Header (Logo & Shop Info + Ticket Number) -->
      <tr>
        <td colspan="3" style="border-right: 1px solid #000; border-bottom: 1px solid #000; padding: 4px 10px; background: #fdfdfd;">
          <div style="display: flex; justify-content: space-between; align-items: center;">
            <img src="https://www.procycling.sk/uploads/image_bank/logo-velocity-removebg-preview.png" style="height: 26px; width: 110px; object-fit: contain; padding-left: 2px;" alt="Velocity">
            <div style="text-align: center; font-size: 10px; line-height: 1.2; flex-grow: 1;">
              <strong style="font-size: 11px;">Medveďovej 1/A, Bratislava-Petržalka</strong><br>
              0917 344 484 &nbsp;|&nbsp; www.velocity.sk
            </div>
          </div>
        </td>
        <td style="border-bottom: 1px solid #000; padding: 4px 10px; background: #f0eaf5;">
          <div style="display: flex; flex-direction: column; justify-content: center; height: 100%;">
            <div style="font-size: 10px; font-weight: 900; text-transform: uppercase; color: #555; margin-bottom: 2px;">Servis č.:</div>
            <div style="font-size: 20px; font-weight: 900; margin: 0 0 2px 0;">${orderNo}</div>
            <div style="font-size: 9px; font-weight: bold; color: #666;">
              Rezervácia: ${resNo ? `<span style="color: #FF0000; font-weight: 900; font-size: 11px;">${resNo}</span>` : '-'}
            </div>
          </div>
        </td>
      </tr>
      
      <!-- Bike Details & Sidebar Setup -->
      <tr>
        <td colspan="3" style="border-right: 1px solid #000; border-bottom: 1px solid #000; padding: 8px 10px; background: #fff;">
          <div style="font-size: 12px; font-weight: 900; text-transform: uppercase; color: #222; letter-spacing: 0.5px; margin-bottom: 6px; text-align: center;">
            Potvrdenie o prebratí do servisu
          </div>
          <div style="background: #f0eaf5; border: 1px solid #d8cedf; border-radius: 6px; padding: 8px 10px;">
            <span style="font-size: 12px; font-weight: 900; color: #444; text-transform: uppercase;">NÁZOV:</span>
            <span style="font-size: 16px; font-weight: 900; margin-left: 6px; color: #000;">${bike}</span>
          </div>
        </td>
        
        <!-- Customer & Service Metadata Sidebar -->
        <td rowspan="${3 + items.length}" style="padding: 0; height: 100%; vertical-align: top; background: #fafafa;">
          <div style="display: flex; flex-direction: column; height: 100%; padding: 6px 8px; box-sizing: border-box;">
            <!-- Customer Panel -->
            <div style="background: #f0eaf5; border: 1px solid #d8cedf; border-radius: 8px; padding: 6px; margin-bottom: 6px;">
              <div style="font-size: 10px; color: #555; font-weight: 900; text-transform: uppercase; margin-bottom: 2px; letter-spacing: 0.5px;">Zákazník</div>
              <div style="font-size: 12px; line-height: 1.3; color: #000;">
                <b style="font-size: 14px; font-weight: 900;">${customer}</b><br>
                <b style="font-weight: 600;">${phone}</b><br>
                <b style="font-weight: 600;">${email}</b>
              </div>
            </div>
            
            <!-- Dates Panel -->
            <div style="background: #f0eaf5; border: 1px solid #d8cedf; border-radius: 8px; padding: 6px; margin-bottom: 6px;">
              <div style="font-size: 10px; color: #555; font-weight: 900; text-transform: uppercase; margin-bottom: 2px; letter-spacing: 0.5px;">Záznam</div>
              <div style="font-size: 12px; line-height: 1.4; color: #000;">
                <span style="color: #444;">Prijatie:</span> <b style="font-weight: 900;">${datumPrijatia}</b><br>
                <span style="color: #444;">Predpokl. odovzdanie:</span> <b style="font-weight: 900;">${datumOdovzdania}</b>
              </div>
              <div style="margin-top: 4px; padding: 3px; background: #fff; border: 1px solid #d8cedf; border-radius: 4px; font-size: 9px; color: #444; text-align: center; line-height: 1.2; white-space: nowrap; overflow: hidden; text-overflow: ellipsis; letter-spacing: -0.1px;">
                O ukončení servisu informujeme cez <b>SMS</b> a <b>e-mail</b>.
              </div>
            </div>
            
            <!-- Status Flag -->
            ${isSpravene ? '<div style="border: 2px solid #28a745; border-radius: 6px; padding: 4px; margin-bottom: 6px; text-align: center; color: #28a745; font-weight: 900; font-size: 14px; background: #fff; text-transform: uppercase; letter-spacing: 1px; box-sizing: border-box;">SPRAVENÉ</div>' : ''}
            
            <!-- Signatures -->
            <div style="${s ? 'margin-top: auto; height: 85px; flex: none;' : 'flex: 1;'} border: 1px solid #bbb; border-radius: 6px; background: #fff; padding: 4px 8px; margin-bottom: 6px; box-sizing: border-box;">
              <div style="font-size: 10px; font-weight: bold; color: #444;">Podpis zákazníka pri odovzdaní do servisu:</div>
            </div>
            <div style="${s ? 'height: 85px; flex: none;' : 'flex: 1;'} border: 1px solid #bbb; border-radius: 6px; background: #fff; padding: 4px 8px; box-sizing: border-box;">
              <div style="font-size: 10px; font-weight: bold; color: #444;">Podpis zákazníka pri prevzatí zo servisu:</div>
            </div>
          </div>
        </td>
      </tr>
      
      <!-- Table Column Headers -->
      <tr>
        <td style="border-right: 1px solid #000; border-bottom: 1px solid #000; padding: 4px 8px; font-weight: bold; font-size: 11px; background: #f4f6f8; text-transform: uppercase; color: #444;">Popis prác:</td>
        <td style="border-right: 1px solid #000; border-bottom: 1px solid #000; padding: 4px; text-align: center; font-weight: bold; font-size: 11px; background: #f4f6f8; text-transform: uppercase; color: #444;">Servis v €</td>
        <td style="border-right: 1px solid #000; border-bottom: 1px solid #000; padding: 4px; text-align: center; font-weight: bold; font-size: 11px; background: #f4f6f8; text-transform: uppercase; color: #444;">Náhr. diely v €</td>
      </tr>
      
      <!-- Service Items Rows -->
      ${items.map(item => `
      <tr>
        <td style="border-right: 1px solid #000; border-bottom: 1px solid #000; padding: 4px 8px; font-size: 12px;">${item.name || '&nbsp;'}</td>
        <td style="border-right: 1px solid #000; border-bottom: 1px solid #000; padding: 4px; font-weight: 600; font-size: 12px; text-align: center;">${item.priceSrv || '&nbsp;'}</td>
        <td style="border-right: 1px solid #000; border-bottom: 1px solid #000; padding: 4px; font-weight: 600; font-size: 12px; text-align: center;">${item.priceDiely || '&nbsp;'}</td>
      </tr>
      `).join('')}
      
      <!-- Total Price Footer -->
      <tr>
        <td style="border-right: 1px solid #000; padding: 8px 10px; font-size: 16px; font-weight: 900; background: #fdfdfd;">SPOLU v EUR</td>
        <td style="border-right: 1px solid #000; padding: 8px; text-align: center; font-size: 16px; font-weight: 900; background: #fdfdfd;">${sumaZakaznik || '&nbsp;'}</td>
        <td style="border-right: 1px solid #000; padding: 8px; text-align: center; vertical-align: middle; background: #fdfdfd;">
          ${resNo ? '<span style="color: #FF0000; font-weight: 900; font-size: 16px; letter-spacing: 1px;">REZ</span>' : ''}
        </td>
      </tr>
    </table>
  </div>
  
  <!-- Right Sidebar Barcode Container (Printed vertically) -->
  <div style="width: 16px; flex-shrink: 0; margin-left: 6px; display: flex; flex-direction: column; justify-content: space-evenly; align-items: center;">
    <img src="https://bwipjs-api.metafloor.com/?bcid=code128&text=${encodeURIComponent(orderNo)}&scale=2&height=6&includetext=false&rotate=R" style="max-height: 48%; max-width: 16px; object-fit: contain;">
    ${resNo ? `<img src="https://bwipjs-api.metafloor.com/?bcid=code128&text=${encodeURIComponent(resNo)}&scale=2&height=6&includetext=false&rotate=R" style="max-height: 48%; max-width: 16px; object-fit: contain;">` : ''}
  </div>
</div>
`;

  // 4. Create or Update Print Overlay
  let overlay = document.getElementById('custom-print-overlay');
  if (!overlay) {
    overlay = document.createElement('div');
    overlay.id = 'custom-print-overlay';
    document.body.appendChild(overlay);
    
    // Add print styles to hide everything else and style the layout
    let style = document.createElement('style');
    style.innerHTML = `
      @media print {
        @page {
          size: A4 portrait;
          margin: 8mm;
        }
        html, body {
          margin: 0 !important;
          padding: 0 !important;
          max-width: 100%;
          background: #fff;
          font-family: Arial, sans-serif;
        }
        body > *:not(#custom-print-overlay) {
          display: none !important;
        }
        #custom-print-overlay {
          display: flex !important;
          flex-direction: column;
          justify-content: space-between;
          position: relative;
          width: 100%;
          min-height: 100vh;
          background: #fff;
          padding: 0;
          margin: 0 auto;
          box-sizing: border-box;
        }
      }
      @media screen {
        #custom-print-overlay {
          display: none !important;
        }
      }
    `;
    document.head.appendChild(style);
  }

  // 5. Populate Overlay & Trigger Print Dialog
  // If single copy is required (sgl=true), generate 1 large ticket.
  // Otherwise, generate 2 identical tickets with a dash split line for easy cutting.
  overlay.innerHTML = sgl 
    ? gen(true) 
    : (gen(false) + '<div style="width: 100%; height: 0; border-bottom: 1px dashed #888; margin: 0.5vh 0;"></div>' + gen(false));

  // Delay printing briefly to ensure resource loading (such as images, barcodes, etc.)
  setTimeout(() => window.print(), 1500);
})();
